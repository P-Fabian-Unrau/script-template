\chapter{Limits}

\begin{definition}[Limit]
  Let $f: D \to \R$ be a function with $D \subseteq \R$ and $a$ a cluster point of $D$. Then $L$ is the limit of $f$ at the point $a$, if for all $\varepsilon > 0$ there exists a $\delta > 0$, such that for all $x \in D$ with $0 < |x - a| < \delta$ we have: $|f(x) - L| < \varepsilon$. We write:
  \[
    \lim_{x \to a} f(x) = L.
  \]
\end{definition}
